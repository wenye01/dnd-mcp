# Milestone 7.5: Client-Server 协同

> **状态**: ⏳ 待开发

> **涉及 D&D 5e 规则**: 否

## 7.5.1 目标

实现 Client 和 Server 的完整协同，支持整局游戏的游玩。

### 背景

Server M7 及 M7.5 功能已完整实现，包括：
- 上下文管理 (`get_context`, `get_raw_context`, `save_message`)
- 状态效果 (`apply_condition`, `remove_condition`, `get_conditions`, `has_condition`)
- 休息系统 (`take_short_rest`, `take_long_rest`, `party_long_rest`)

Client 需要重构以适配 Server 设计，实现数据流：

```
用户输入 -> Client 获取上下文 -> LLM 决策 -> 调用 MCP Tool -> 规则引擎执行 -> 保存对话 -> 返回结果
```

### 关键变更

| 变更项 | 当前状态 | 目标状态 |
|--------|----------|----------|
| 对话历史存储 | Client 本地存储 (Redis/PostgreSQL) | Server 统一存储 |
| 上下文构建 | Client 从本地加载 | Client 调用 Server `get_context` |
| 消息保存 | Client 本地保存 | Client 调用 Server `save_message` |
| 术语 | Session | Campaign |

// packages/client/internal/server/client.go

// ServerClient Server API 客户端接口
type ServerClient interface {
    // 上下文管理
    GetContext(ctx context.Context, campaignID string, limit int, includeCombat bool) (*Context, error)
    GetRawContext(ctx context.Context, campaignID string) (*RawContext, error)

    // MCP Tool 调用（复用现有 mcp.MCPClient）

}

// Context 压缩后的上下文
type Context struct {
    CampaignID      string        `json:"campaign_id"`
    GameSummary     *GameSummary  `json:"game_summary"`
    Messages        []Message     `json:"messages"`
    RawMessageCount int           `json:"raw_message_count"`
    TokenEstimate   int           `json:"token_estimate"`
}

// RawContext 原始上下文（完整模式）
type RawContext struct {
    CampaignID   string       `json:"campaign_id"`
    GameState    *GameState   `json:"game_state"`
    Characters   []*Character `json:"characters"`
    Combat       *Combat      `json:"combat"`
    Map          *Map         `json:"map"`
    Messages     []*Message   `json:"messages"`
    MessageCount int          `json:"message_count"`
}
```

// packages/client/internal/service/context_builder.go

type ContextBuilder struct {
    serverClient  server.ServerClient  // Server API 客户端
    useRawContext bool                  // 是否使用原始上下文
}

func NewContextBuilder(serverClient server.ServerClient) *ContextBuilder {
    return &ContextBuilder{
        serverClient:  serverClient,
        useRawContext: false, // 默认使用简化模式
    }
}

func (b *ContextBuilder) BuildContext(ctx context.Context, campaignID, userMessage string) ([]llm.Message, error) {
    // 1. 从 Server 获取上下文
    var serverCtx *server.Context
    var err error

    if b.useRawContext {
        // 完整模式：获取原始数据
        rawCtx, err := b.serverClient.GetRawContext(ctx, campaignID)
        if err != nil {
            return nil, err
        }
        serverCtx = b.compressContext(rawCtx)
    } else {
        // 简化模式：使用 Server 压缩后的上下文
        serverCtx, err = b.serverClient.GetContext(ctx, campaignID, 20, true)
        if err != nil {
            return nil, err
        }
    }

    // 2. 构建 System 消息
    messages := []llm.Message{
        {
            Role:    "system",
            Content: b.buildSystemPrompt(serverCtx),
        },
    }

    // 3. 添加历史消息
    for _, msg := range serverCtx.Messages {
        messages = append(messages, llm.Message{
            Role:    string(msg.Role),
            Content: msg.Content,
        })
    }

    // 4. 添加当前用户消息
    messages = append(messages, llm.Message{
        Role:    "user",
        Content: userMessage,
    })

    return messages, nil
}

func (b *ContextBuilder) buildSystemPrompt(ctx *server.Context) string {

    // ...
}

// packages/client/internal/service/chat.go

type ChatService struct {
    serverClient   server.ServerClient  // Server API 客户端
    llmClient      llm.LLMClient
    contextBuilder *ContextBuilder
}

func NewChatService(
    serverClient server.ServerClient,
    llmClient llm.LLMClient,
    contextBuilder *ContextBuilder,
) *ChatService {
    return &ChatService{
        serverClient:   serverClient,
        llmClient:      llmClient,
        contextBuilder: contextBuilder,
    }
}

func (s *ChatService) SendMessage(ctx context.Context, campaignID string, req *SendMessageRequest) (*Message, error) {
    // 1. 保存用户消息到 Server
    userMsg := &server.Message{
        CampaignID: campaignID,
        Role:       "user",
        Content:    req.Content,
        PlayerID:   req.PlayerID,
    }
    if err := s.serverClient.SaveMessage(ctx, campaignID, userMsg); err != nil {
        return nil, err
    }

    // 2. 构建上下文（从 Server 获取）
    messages, err := s.contextBuilder.BuildContext(ctx, campaignID, req.Content)
    if err != nil {
        return nil, err
    }

    // 3. 调用 LLM
    llmResp, err := s.llmClient.Chat(ctx, &llm.ChatRequest{Messages: messages})
    if err != nil {
        return nil, err
    }

    // 4. 处理响应
    // ...

    // 5. 保存助手消息到 Server
    assistantMsg := &server.Message{
        CampaignID: campaignID,
        Role:       "assistant",
        Content:    llmResp.Choices[0].Message.Content,
    }
    if err := s.serverClient.SaveMessage(ctx, campaignID, assistantMsg); err != nil {
        return nil, err
    }

    return assistantMsg, nil
}
```
