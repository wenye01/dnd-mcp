## 1. MCP 协议对接

### 1.1 连接方式

Server 通过 HTTP 暴露 MCP 协议端点：

| 端点 | 方法 | 描述 |
|------|------|------|
| `/health` | GET | 健康检查 |
| `/mcp/initialize` | POST | MCP 初始化握手 |
| `/mcp/tools` | GET | 获取可用工具列表 |
| `/mcp/tools/call` | POST | 调用工具 |

### 1.2 工具调用流程

```
Client                                      Server (MCP)
  │                                           │
  │  1. Initialize                            │
  │  POST /mcp/initialize                     │
  │ ───────────────────────────────────────> │
  │                                           │
  │  2. Initialize Response                   │
  │  { protocolVersion, capabilities, tools } │
  │ <──────────────────────────────────────── │
  │                                           │
  │  3. List Tools                            │
  │  GET /mcp/tools                           │
  │ ───────────────────────────────────────> │
  │                                           │
  │  4. Tools List                            │
  │  [{ name, description, inputSchema }]     │
  │ <──────────────────────────────────────── │
  │                                           │
  │  5. Call Tool                             │
  │  POST /mcp/tools/call                     │
  │  { name: "attack", arguments: {...} }     │
  │ ───────────────────────────────────────> │
  │                                           │
  │  6. Tool Result                           │
  │  { content: [{ type, text }], isError }   │
  │ <──────────────────────────────────────── │
```

## 2. GLM-4.7-Flash 集成

### 2.1 API 调用封装

```go
// packages/client/internal/llm/glm.go

package llm

type GLMClient struct {
    apiKey     string
    baseURL    string
    httpClient *http.Client
}

type ChatMessage struct {
    Role         string     `json:"role"`
    Content      string     `json:"content"`
    ToolCalls    []ToolCall `json:"tool_calls,omitempty"`
    ToolCallID   string     `json:"tool_call_id,omitempty"`
}

type ToolCall struct {
    ID       string       `json:"id"`
    Type     string       `json:"type"`
    Function FunctionCall `json:"function"`
}

type FunctionCall struct {
    Name      string `json:"name"`
    Arguments string `json:"arguments"`
}

type Tool struct {
    Type     string             `json:"type"`
    Function FunctionDefinition `json:"function"`
}

type FunctionDefinition struct {
    Name        string                 `json:"name"`
    Description string                 `json:"description"`
    Parameters  map[string]interface{} `json:"parameters"`
}

type ChatRequest struct {
    Model       string        `json:"model"`
    Messages    []ChatMessage `json:"messages"`
    Tools       []Tool        `json:"tools,omitempty"`
    ToolChoice  interface{}   `json:"tool_choice,omitempty"`
    Stream      bool          `json:"stream"`
    Temperature float64       `json:"temperature"`
    MaxTokens   int           `json:"max_tokens"`
}

type ChatResponse struct {
    ID      string    `json:"id"`
    Object  string    `json:"object"`
    Created int64     `json:"created"`
    Model   string    `json:"model"`
    Choices []Choice  `json:"choices"`
    Usage   Usage     `json:"usage"`
}

type Choice struct {
    Index        int         `json:"index"`
    Message      ChatMessage `json:"message"`
    FinishReason string      `json:"finish_reason"`
}

type Usage struct {
    PromptTokens     int `json:"prompt_tokens"`
    CompletionTokens int `json:"completion_tokens"`
    TotalTokens      int `json:"total_tokens"`
}
```

### 2.2 Tools 参数格式转换

```go
// MCP Tools 到 GLM 格式转换
func MCPToolToGLMTool(mcpTool mcp.Tool) Tool {
    return Tool{
        Type: "function",
        Function: FunctionDefinition{
            Name:        mcpTool.Name,
            Description: mcpTool.Description,
            Parameters: map[string]interface{}{
                "type":       mcpTool.InputSchema.Type,
                "properties": mcpTool.InputSchema.Properties,
                "required":   mcpTool.InputSchema.Required,
            },
        },
    }
}
```

### 2.3 流式响应处理

```go
func (c *GLMClient) ChatStream(ctx context.Context, req *ChatRequest) (<-chan ChatChunk, <-chan error) {
    chunkChan := make(chan ChatChunk)
    errChan := make(chan error)

    go func() {
        defer close(chunkChan)
        defer close(errChan)

        // SSE 解析逻辑
        scanner := bufio.NewScanner(resp.Body)
        for scanner.Scan() {
            line := scanner.Text()
            if strings.HasPrefix(line, "data: ") {
                data := strings.TrimPrefix(line, "data: ")
                if data == "[DONE]" {
                    break
                }
                var chunk ChatChunk
                if err := json.Unmarshal([]byte(data), &chunk); err == nil {
                    chunkChan <- chunk
                }
            }
        }
    }()

    return chunkChan, errChan
}
```

### 4.2 对话编排服务

```go
// packages/client/internal/service/orchestrator.go

type Orchestrator struct {
    llmClient   *glm.GLMClient
    mcpClient   *mcp.Client
    maxTurns    int
}

func (o *Orchestrator) ProcessMessage(ctx context.Context, sessionID, userMessage string) (*Response, error) {
    // 1. 获取上下文
    messages := o.buildContext(ctx, sessionID)

    // 2. 添加用户消息
    messages = append(messages, glm.ChatMessage{
        Role:    "user",
        Content: userMessage,
    })

    // 3. 获取工具列表
    tools := o.getLLMTools(ctx)

    // 4. 处理对话（可能包含多轮工具调用）
    return o.processWithTools(ctx, messages, tools)
}

func (o *Orchestrator) processWithTools(ctx context.Context, messages []glm.ChatMessage, tools []glm.Tool) (*Response, error) {
    for turn := 0; turn < o.maxTurns; turn++ {
        // 调用 LLM
        resp, err := o.llmClient.Chat(ctx, &glm.ChatRequest{
            Model:    "glm-4.7-flash",
            Messages: messages,
            Tools:    tools,
        })
        if err != nil {
            return nil, err
        }

        // 检查是否有工具调用
        if len(resp.Choices[0].Message.ToolCalls) == 0 {
            return &Response{Content: resp.Choices[0].Message.Content}, nil
        }

        // 执行工具调用
        messages = append(messages, resp.Choices[0].Message)
        for _, call := range resp.Choices[0].Message.ToolCalls {
            result, _ := o.mcpClient.CallTool(ctx, call.Function.Name, parseArgs(call.Function.Arguments))
            messages = append(messages, glm.ChatMessage{
                Role:       "tool",
                ToolCallID: call.ID,
                Content:    result.Content,
            })
        }
    }

    return nil, errors.New("max turns exceeded")
}
```

## 5. 错误处理

### 5.1 API 错误码

| 错误码 | HTTP状态 | 描述 |
|--------|----------|------|
| `ERR_MCP_CONNECTION` | 503 | 无法连接到 MCP Server |
| `ERR_MCP_TIMEOUT` | 504 | MCP 调用超时 |
| `ERR_MCP_TOOL_NOT_FOUND` | 400 | 工具不存在 |
| `ERR_MCP_INVALID_ARGUMENTS` | 400 | 工具参数无效 |
| `ERR_LLM_API_ERROR` | 502 | LLM API 调用失败 |
| `ERR_CONTEXT_TOO_LONG` | 400 | 上下文超长 |
