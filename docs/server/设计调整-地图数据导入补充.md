### 3.2 领域模型视角 (Architect-Domain)

**新增模型定义**:

```go
// ImportFormat 导入格式
type ImportFormat string

const (
    ImportFormatUVTT       ImportFormat = "uvtt"
    ImportFormatFVTTScene  ImportFormat = "fvtt_scene"
    ImportFormatFVTTModule ImportFormat = "fvtt_module"
)

// ImportMeta 导入元数据（扩展 M6 定义）
type ImportMeta struct {
    Format       ImportFormat `json:"format"`
    OriginalID   string       `json:"original_id,omitempty"`
    OriginalName string       `json:"original_name,omitempty"`
    ModuleID     string       `json:"module_id,omitempty"`     // FVTT 模块 ID
    ImportedAt   time.Time    `json:"imported_at"`
    RawJSON      string       `json:"raw_json,omitempty"`      // 压缩 JSON
}

// ImportOptions 导入选项
type ImportOptions struct {
    Format          ImportFormat `json:"format,omitempty"`           // 强制指定格式
    Name            string       `json:"name,omitempty"`             // 覆盖名称
    ImportTokens    bool         `json:"import_tokens"`              // 是否导入 Token
    ImportWalls     bool         `json:"import_walls"`               // 是否导入墙体
    ImageStorage    string       `json:"image_storage,omitempty"`    // base64/url/file
    MaxImageSize    int64        `json:"max_image_size,omitempty"`   // 最大图片大小 (bytes)
}

// ImportResult 导入结果
type ImportResult struct {
    Map       *Map      `json:"map"`
    Warnings  []string  `json:"warnings"`
    Skipped   []string  `json:"skipped"`    // 跳过的字段
}
```
