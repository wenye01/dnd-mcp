#### 技术细节

**环境配置文件**:

```bash
# .env.example

# 数据库配置
REDIS_HOST=localhost:6379
DATABASE_URL=postgres://postgres:postgres@localhost:5432/dnd_mcp?sslmode=disable

# GLM-4.7-Flash 配置
GLM_API_KEY=your-api-key-here
GLM_API_ENDPOINT=https://open.bigmodel.cn/api/paas/v4/chat/completions
GLM_MODEL=glm-4.7-flash
GLM_MAX_TOKENS=65536

# Client 配置
HTTP_HOST=0.0.0.0
HTTP_PORT=8080

# Server 配置 (MCP)
MCP_SERVER_COMMAND=./bin/dnd-server.exe
```

**启动脚本**:

```powershell
# scripts/start-all.ps1

# 启动完整系统

Write-Host "Starting DND MCP System..."

# 1. 检查 Redis
Write-Host "Checking Redis..."
redis-cli ping

# 2. 检查 PostgreSQL
Write-Host "Checking PostgreSQL..."

# 3. 构建 Server
Write-Host "Building Server..."
go build -o bin/dnd-server.exe ./packages/server/cmd/server

# 4. 构建 Client
Write-Host "Building Client..."
go build -o bin/dnd-client.exe ./packages/client/cmd/api

# 5. 启动 Client (会自动启动 Server)
Write-Host "Starting Client..."
.\bin\dnd-client.exe

Write-Host "System started on http://localhost:8080"
```

**Docker Compose** (可选):

```yaml
# docker-compose.yml

version: '3.8'

services:
  redis:
    image: redis:7-alpine
    ports:
      - "6379:6379"

  postgres:

    environment:
      POSTGRES_DB: dnd_mcp
      POSTGRES_USER: postgres
      POSTGRES_PASSWORD: postgres
    ports:
      - "5432:5432"

  # Server 和 Client 需要在本地构建运行
```

```go
// tests/e2e/integration_test.go

package e2e

import (
    "context"
    "net/http"
    "testing"
    "time"

    "github.com/stretchr/testify/assert"
    "github.com/stretchr/testify/require"
)

func TestE2E_FullGameFlow(t *testing.T) {
    // 前置条件: Server 和 Client 已启动

    client := &http.Client{Timeout: 30 * time.Second}
    baseURL := "http://localhost:8080"

    t.Run("创建会话", func(t *testing.T) {
        resp, err := client.Post(baseURL+"/api/sessions", "application/json",
            strings.NewReader(`{"name":"测试战役"}`))
        require.NoError(t, err)
        defer resp.Body.Close()

        assert.Equal(t, 201, resp.StatusCode)
    })

    t.Run("创建角色", func(t *testing.T) {
        // 发送消息让 AI 创建角色
        resp, err := client.Post(baseURL+"/api/sessions/"+sessionID+"/chat",
            "application/json",
            strings.NewReader(`{"content":"我想创建一个精灵游侠，名字叫Legolas"}`))
        require.NoError(t, err)
        defer resp.Body.Close()

        // 验证响应包含角色创建信息
        // 可能需要多轮对话
    })

    t.Run("投掷骰子", func(t *testing.T) {
        resp, err := client.Post(baseURL+"/api/sessions/"+sessionID+"/chat",
            "application/json",
            strings.NewReader(`{"content":"我投一个d20"}`))
        require.NoError(t, err)

        // 验证响应包含骰子结果
    })

    t.Run("战斗流程", func(t *testing.T) {
        // 开始战斗
        resp, err := client.Post(baseURL+"/api/sessions/"+sessionID+"/chat",
            "application/json",
            strings.NewReader(`{"content":"我想和哥布林战斗"}`))
        require.NoError(t, err)

        // 攻击
        resp, err = client.Post(baseURL+"/api/sessions/"+sessionID+"/chat",
            "application/json",
            strings.NewReader(`{"content":"我攻击哥布林"}`))
        require.NoError(t, err)

        // 结束战斗
    })
}
```
